8def _fix_length(output: Iterable[str]) -> str: 

9 """Fixes the length of the circuit. 

10 

11 Args: 

12 output (str): The circuit that is to be printed out. 

13 

14 Returns: 

15 str: Circuit that is realigned. 

16 

17 Qubit 0: ---|H|- 

18 Qubit 1: - 

19 

20 becomes: 

21 

22 Qubit 0: ---|H|- 

23 Qubit 1: ------- 

24 

25 """ 

26 for i, val in enumerate(output): 

27 if len(val) < len(max(output, key=len)): 

28 diff = len(max(output, key=len)) - len(val)
