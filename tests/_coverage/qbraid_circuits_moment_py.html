25class Moment: 

26 """ 

27 A time-slice of instructions within a circuit. 

28 Grouping instructions into moments is intended to be a strong suggestion to 

29 whatever is scheduling instructions on real hardware. instructions in the same 

30 moment should execute at the same time (to the extent possible; not all 

31 instructions have the same duration) and it is expected that all instructions 

32 in a moment should be completed before beginning the next moment. 

33 Moment can be indexed by qubit or list of qubits: 

34 """ 

35 

36 def __init__(self, instructions: Union[Instruction, Iterable[Instruction]] = None): 

37 if instructions is None: 

38 self._instructions = [] 

39 elif isinstance(instructions, Iterable): 

40 self._instructions = instructions 

41 else: 

42 self._instructions = [instructions] 

43 

44 @property 

45 def instructions(self): 

46 return self._instructions 

47 

48 @property 

49 def qubits(self) -> Iterable[Qubit]: 

50 out = set() 

51 for instruction in self._instructions: 

52 out.update(instruction.qubits) 

53 return list(out) 

54 

55 def __repr__(self): 

56 return f'Moment("{self.instructions}")' 

57 

58 def appendable(self, instruction: Instruction) -> bool: 

59 """Checks that the instructions are disjoint from the qubits acted on in the moment. 

60 

61 Args: 

62 instruction (Instruction): Instruction that needs to be checked. 

63 

64 Returns: 

65 bool: True if disjoint, false otherwise. 

66 """ 

67 return set(instruction.qubits).isdisjoint(self.qubits)

69 def append(self, instruction: Union[Instruction, Iterable[Instruction], Set]) -> None: 

70 

71 """Wrapper which preps instructions to be appended, throws error if not appendable. 

72 

73 Args: 

74 instruction (Union[Instruction, Iterable[Instruction], Set]): 

75 The instruction or iterable of instructions to append. 

76 

77 Raises: 

78 TypeError: Non appendable type. 

79 """ 

80 # TODO: add generator appending functionality. 

81 if isinstance(instruction, Instruction): 

82 self._insert(instruction) 

83 # might be redundant to have iterable here since circuit 

84 # only appends Instructions or Moments. 

85 elif isinstance(instruction, Iterable): 

86 for i in instruction: 

87 self.append(i) 

88 # error 

89 else: 

90 raise TypeError("Instructions of type {} not appendable".format(type(instruction))) 

91 

92 def _insert(self, instruction: Instruction) -> None: 

93 """ 

94 Appends instructions onto the end of the moment 

95 Args: 

96 instruction (Instruction): Operations that are applied to qubit(s) in a circuit. 

97 

98 Raises: 

99 CircuitError: Errors that occur when constructing circuits with the qBraid circuit layer. 

100 """ 

101 if self.appendable(instruction): 

102 self._instructions.append(instruction) 

103 else: 

104 raise CircuitError("Instructions of type {} not appendable".format(type(instruction)))
