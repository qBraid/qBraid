14def mask_value(value): 

15 if value is None: 

16 return "None" 

17 val = round(len(value) / 5) 

18 len_hint = 4 if val > 4 else 0 if val < 2 else val 

19 default_stars = len(value) - len_hint 

20 len_stars = default_stars if default_stars <= 16 else 16 

21 return ("*" * len_stars) + value[-len_hint:]
